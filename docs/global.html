Indicates whether the column values must be unique.

Indicates whether the column is a primary key.

Example
const tableSchema = {
    tableName: 'tableName',
    dbSchema: 'schemaName',
    timeStamps: true,
    useCS: false,
    columns: [
        {
            name: 'fieldName',
            type: 'dataType',
            length: 255,
            unique: false,
            notNull: false,
            primary: false,
            default: 'value',
            useDefault: false,
        },
    ],
    foreignKeys: [
        {
            hasRelations: [{ name: 'field' }],
            withColumns: [{ name: 'withField' }],
            withTable: 'withTable',
            onDeleteAction: 'action',
            onUpdateAction: 'action',
        },
    ],
};
